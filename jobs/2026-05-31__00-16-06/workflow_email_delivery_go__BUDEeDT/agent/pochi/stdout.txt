package main

import (
	"context"
	"fmt"
	"log"
	"os"

	"github.com/magicbell/magicbell-go"
	"github.com/magicbell/magicbell-go/client"
	"github.com/magicbell/magicbell-go/clientconfig"
)

func main() {
	token := os.Getenv("MAGICBELL_PROJECT_TOKEN")
	runID := os.Getenv("ZEALT_RUN_ID")
	gmailUser := os.Getenv("GMAIL_USER_NAME")

	if token == "" || runID == "" || gmailUser == "" {
		log.Fatal("Missing environment variables")
	}

	workflowKey := fmt.Sprintf("order_notification_%s", runID)
	email := fmt.Sprintf("%s+%s@gmail.com", gmailUser, runID)

	config := clientconfig.NewConfig()
	config.SetAccessToken(token)
	
	mbClient := client.NewClient(config)

	saveReq := magicbell.SaveWorkflowRequest{
		Key: workflowKey,
		Steps: []magicbell.WorkflowDefinitionSteps{
			{
				Command: "broadcast",
				Input: map[string]interface{}{
					"title":   "Order Notification - {{payload.run_id}}",
					"content": "Your order has been updated successfully. Run ID: {{payload.run_id}}",
					"recipients": []map[string]interface{}{
						{"email": "{{payload.email}}"},
					},
				},
			},
		},
	}

	_, err := mbClient.Workflows.SaveWorkflow(context.Background(), workflowKey, saveReq)
	if err != nil {
		log.Fatalf("Failed to save workflow: %v", err)
	}

	triggerReq := magicbell.ExecuteWorkflowRequest{
		Input: map[string]interface{}{
			"run_id": runID,
			"email":  email,
		},
	}

	runRes, err := mbClient.Workflows.CreateWorkflowRun(context.Background(), workflowKey, triggerReq)
	if err != nil {
		log.Fatalf("Failed to trigger workflow: %v", err)
	}

	fmt.Printf("Workflow Run ID: %s\n", runRes.Id)
}
