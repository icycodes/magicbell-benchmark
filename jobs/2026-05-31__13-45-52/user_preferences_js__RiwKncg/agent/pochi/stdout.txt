const jwt = require('jsonwebtoken');

const runId = process.env.ZEALT_RUN_ID;
const apiKey = process.env.MAGICBELL_API_KEY;
const secretKey = process.env.MAGICBELL_SECRET_KEY;

const userEmail = 'user-' + runId + '@example.com';
const userExternalId = 'user_' + runId;

const tokenPayload = {
  user_email: userEmail,
  user_external_id: userExternalId,
  api_key: apiKey,
};

const userJwt = jwt.sign(tokenPayload, secretKey, { algorithm: 'HS256' });

// First, fetch current preferences
fetch('https://api.magicbell.com/v2/channels/user_preferences', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer ' + userJwt,
    'Content-Type': 'application/json',
  },
}).then(r => r.json().then(t => {
  console.log('GET preferences:', JSON.stringify(t, null, 2));
  
  // Now try to save preferences with billing category
  return fetch('https://api.magicbell.com/v2/channels/user_preferences', {
    method: 'PUT',
    headers: {
      'Authorization': 'Bearer ' + userJwt,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      categories: [
        { key: 'comments', channels: [{ name: 'in_app', enabled: true }, { name: 'email', enabled: true }] },
        { key: 'billing', channels: [{ name: 'in_app', enabled: true }, { name: 'email', enabled: false }] },

    }),
  });
})).then(r => {

    },
  });
