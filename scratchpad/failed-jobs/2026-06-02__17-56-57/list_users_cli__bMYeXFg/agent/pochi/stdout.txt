import json
import os
import subprocess

external_id = f"list-users-cli-{os.environ['ZEALT_RUN_ID']}"
starting_after = None
found = None

while True:
    cmd = ["magicbell", "user", "list", "--limit", "50"]
    if starting_after:
        cmd += ["--starting_after", starting_after]
    result = subprocess.run(cmd, capture_output=True, text=True)
    if result.returncode != 0:
        raise SystemExit(result.stderr)
    payload = json.loads(result.stdout)
    for user in payload.get("data", []):
        if user.get("external_id") == external_id:
            found = user
            break
    if found:
        break
    next_link = payload.get("links", {}).get("next")
    if not next_link:
        break
    # extract starting_after param
    if "starting_after=" in next_link:
        starting_after = next_link.split("starting_after=")[-1]
    else:
        break

if not found:
    raise SystemExit(f"User with external_id {external_id} not found")

print(json.dumps({"id": found.get("id"), "external_id": found.get("external_id")}))
