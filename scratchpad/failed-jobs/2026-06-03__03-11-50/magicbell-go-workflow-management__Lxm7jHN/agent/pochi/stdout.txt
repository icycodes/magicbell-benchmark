type CustomHook struct {

	action      string
}

func (h *CustomHook) BeforeRequest(req hooks.Request, params map[string]string) hooks.Request {
	if req.GetPath() == "/workflows/*" {
		req.SetPath(fmt.Sprintf("/workflows/%s", h.workflowKey))
		if h.action == "delete" {
			req.SetMethod("DELETE")
		}
	}
	return req
}

func (h *CustomHook) AfterResponse(req hooks.Request, resp hooks.Response, params map[string]string) hooks.Response {
